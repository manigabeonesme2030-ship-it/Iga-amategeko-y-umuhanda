function login(){const name=document.getElementById("loginName").value.trim();const pass=document.getElementById("loginPass").value.trim();if(name===adminName && pass===adminPass){document.getElementById("loginView").style.display="none";document.getElementById("adminView").style.display="block";} else {document.getElementById("loginView").style.display="none";document.getElementById("userView").style.display="block";}}
function addQuestion(){const val=document.getElementById("newQuestion").value.trim();const parts=val.split("|");if(parts.length!==6){alert("Format ya question ntikwiye!"); return;}questions.push({q:parts, options:[parts[1],parts[2],parts[3],parts[4]], answer:parseInt(parts[5])});document.getElementById("adminStatus").innerText=`Questions: ${questions.length}`;document.getElementById("newQuestion").value="";}
function addCode(){const code=document.getElementById("newCode").value.trim();const uses=parseInt(document.getElementById("codeUses").value);if(!code||!uses){alert("Andika code na uses!"); return;}codes.push({code:code, remainingUses:uses});document.getElementById("adminStatus").innerText=`Codes: ${codes.length}`;document.getElementById("newCode").value="";document.getElementById("codeUses").value="";}
function saveNotes(){notes=document.getElementById("notesArea").value.trim();document.getElementById("adminStatus").innerText="Notes & links saved!";}
function saveSocial(){socialLinks.fb=document.getElementById("fbLink").value.trim();socialLinks.tiktok=document.getElementById("tiktokLink").value.trim();document.getElementById("adminStatus").innerText="Social Links saved!";}
function verifyCode(){const enteredCode=document.getElementById("userCode").value.trim();const found=codes.find(c=>c.code===enteredCode && c.remainingUses>0);if(!found){alert("Code ntabwo iri valid cyangwa yamaze gukoreshwa!"); return;}found.remainingUses--;alert("Code yemerewe! Tangira exam.");document.getElementById("paymentPrompt").style.display="none";document.getElementById("examDiv").style.display="block";document.getElementById("timerDiv").style.display="block";startTimer(20);loadExam();}
function startTimer(minutes){let time=minutes*60;timerInterval=setInterval(()=>{let min=Math.floor(time/60);let sec=time%60;document.getElementById("timer").innerText=`Iminota isigaye: ${min.toString().padStart(2,'0')}:${sec.toString().padStart(2,'0')}`;if(time<=0){clearInterval(timerInterval); submitExam(); alert("Igihe kirarangiye! Exam irarangiye.");}time--;},1000);}
function loadExam(){selectedQuestions=[];const usedIndices=new Set();while(selectedQuestions.length<20){let idx=Math.floor(Math.random()*questions.length);if(!usedIndices.has(idx)){selectedQuestions.push(questions[idx]); usedIndices.add(idx);}if(usedIndices.size===questions.length) break;}const container=document.getElementById("questionsContainer");container.innerHTML="";selectedQuestions.forEach((q,i)=>{let div=document.createElement("div");div.className="question";div.innerHTML=`<p>${i+1}. ${q.q}</p>`+ q.options.map((opt,j)=>`<label><input type="radio" name="q${i}" value="${j}"> ${opt}</label><br>`).join("");container.appendChild(div);});}
function submitExam(){clearInterval(timerInterval);let score=0;selectedQuestions.forEach((q,i)=>{const radios=document.getElementsByName("q"+i);radios.forEach(r=>{if(r.checked && parseInt(r.value)===q.answer) score++;});});document.getElementById("scoreDiv").innerHTML=`<h3>Watsinze: ${score}/${selectedQuestions.length}</h3>`;}
